const { createApp } = Vue; createApp({ data() { return { dogImage:'' }; }, methods:{ async fetchDogImages() { try { const res = await fetch ('https://dog.ceo/api/breeds/image/random'); const data = await res.json(); this.dogImage = data.image; } catch { this.dogImage = 'Failed to fetch images'; } } } , mounted(){ this.fetchDogImages(); } }).mount('#app');
